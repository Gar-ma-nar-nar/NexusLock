export function cosine(a: number[], b: number[]): number {
  if (a.length !== b.length || a.length === 0) return 0;
  let dot = 0, na = 0, nb = 0;
  for (let i = 0; i < a.length; i++) { dot += a[i]*b[i]; na += a[i]*a[i]; nb += b[i]*b[i]; }
  const d = Math.sqrt(na) * Math.sqrt(nb);
  return d === 0 ? 0 : dot / d;
}

export function bestMatch(sig: Embedding, faces: SavedFace[], threshold: number) {
  let best = null;
  for (const f of faces) {
    if (f.embedding.length !== sig.length) continue;
    const s = cosine(sig, f.embedding);
    if (!best || s > best.score) best = { face: f, score: s };
  }
  return best && best.score >= threshold ? best : null;
}

export function l2Normalize(v: number[]): number[] {
  const n = Math.sqrt(v.reduce((s, x) => s + x*x, 0));
  return n === 0 ? v.slice() : v.map(x => x / n);
}

class FaceNetEngine {
  private model: any = null;
  async ensureLoaded() {
    if (this.model) return true;
    try {
      const modelAsset = require('./assets/models/facenet.tflite');
      this.model = await loadTensorflowModel(modelAsset);
      return true;
    } catch { return false; }
  }
  async embed(buffer: any): Promise<Embedding | null> {
    if (!(await this.ensureLoaded()) || !this.model) return null;
    try {
      const [out] = await this.model.run([buffer]);
      return l2Normalize(Array.from(new Float32Array(out)));
    } catch { return null; }
  }
}
